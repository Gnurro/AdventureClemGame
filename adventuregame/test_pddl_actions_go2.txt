(:action GO
    :parameters (?c ?d - room ?p - player ?i - inventory)
    :precondition (and
        (at ?p ?c)
        (exit ?c ?d)
        (not (at ?p ?d))
        )
    :effect (and
        (at ?p ?d)
        (not (at ?p ?c))
        (forall (?e - takeable)
            (and
                (taken ?e ?i)
                (brought ?e ?i)
                (when
                    (in ?e ?i)
                    (and
                        (at ?e ?d)
                        (not (at ?e ?c))
                    )
                )
            )
        )
    )
)
